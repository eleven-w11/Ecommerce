const renderProductItem = (product) => {
    const imageObj = product.images?.[0];
    if (!imageObj) return null;

    const imageKey = Object.keys(imageObj).find((key) =>
        key.startsWith("pi_")
    );
    const imageToShow = imageObj[imageKey];

    // Get match source info (backend se aya hua)
    const matchInfo = product.matchSource?.length
        ? `Matched in: ${product.matchSource.join(", ")}`
        : "";

    return (
        <Link
            key={product._id}
            to={`/product/${product._id}`}
            className="search-item"
            onClick={onClose}
        >
            <img
                src={`/images/${imageToShow}`}
                alt={product.product_name}
                className="search-item-image"
                onError={(e) => {
                    e.target.src = "/images/placeholder.jpg";
                }}
            />
            <div className="search-item-details">
                <h3>{product.product_name}</h3>
                <p className="search-item-price">
                    ${(product.product_price || 0).toFixed(2)}
                </p>
                {matchInfo && (
                    <p className="search-item-match">{matchInfo}</p>
                )}
            </div>
        </Link>
    );
};
